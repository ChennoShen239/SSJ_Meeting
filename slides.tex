\section{Why Sequence Space?}

\begin{frame}{Sequence Space vs.\ State Space}{Motivation}

    \begin{itemize}
        \item In heterogeneous-agent (HA) models, the economy’s state is the \alert{entire cross-sectional distribution} — an infinite-dimensional object.
        \item Sequence-space methods \emph{sidestep} this challenge: they characterize \alert{local dynamics around the steady state} without ever carrying the distribution as a state variable.
        \item The key insight: aggregate equilibrium responses to shocks can be encoded in \alert{Jacobian matrices} of aggregate outcomes with respect to aggregate inputs.
    \end{itemize}

\end{frame}


\begin{frame}{Sequence Space vs.\ State Space}{State Space}

    \begin{definition}
        A \alert{state-space representation} describes the economy recursively.
        At date $t$, a state variable $x_t$ summarizes all payoff-relevant history, and the model evolves as
        \[
            x_{t+1}=f(x_t,\varepsilon_{t+1}),
            \qquad
            y_t=g(x_t).
        \]
        \emph{Given $x_t$, the future is independent of the past.}
    \end{definition}
\end{frame}

\begin{frame}{Sequence Space vs.\ State Space}{Sequence Space}

    \begin{definition}
        A \alert{sequence-space representation} characterizes equilibrium as a mapping between entire time paths. We seek sequences $\{X_t\}_{t=0}^{\infty}$ such that equilibrium conditions hold at every date:
        \[
            F\!\left(\{X_t\}_{t=0}^{\infty},\{Z_t\}_{t=0}^{\infty}\right)=\bm{0},
        \]
        where $\{X_t\}$ collects endogenous variables and $\{Z_t\}$ collects exogenous shocks or policy paths.
    \end{definition}

\end{frame}


\begin{frame}{Sequence Space vs.\ State Space}{Comparison}

    \begin{block}{Intuition}
        \begin{itemize}
            \item State space asks: \emph{what is the current state, and how does it transition?}
            \item Sequence space asks: \emph{given a shock path, what equilibrium outcome path clears all markets?}
        \end{itemize}
    \end{block}

    \begin{block}{Takeaway}
        When the distribution is the intractable state variable, sequence space reframes a ``huge state'' problem as a \alert{structured Jacobian} problem — amenable to fast, scalable computation.
    \end{block}
\end{frame}



\section{Local Solution}
\subsection{SSJ in Krusell-Smith Model}
\begin{frame}{Example: The Krusell-Smith Model}{Setup}
    \begin{itemize}
        \item The economy is populated by a mass 1 of heterogeneous households,
        \item Each household maximizes its expected lifetime utility:
              \[
                  \mathbb{E}_0\left[\sum_{t=0}^{\infty} \beta^t u(c_t)\right],
              \]
        \item Standard CRRA utility,
        \item There exist $n_e$ idiosyncratic states w/ transition matrix $P(e,e')$ and stationary distribution $\pi$,
        \item Agents supply an exogenous number of hours $n$, and earn wage income $w_t e n$,
        \item Agents can only trade in capital $k$, which pays a rental rate $r_t$ net of depreciation.
    \end{itemize}


\end{frame}

\begin{frame}{Example: The Krusell-Smith Model}{Household Problem}
    The value function of an agent entering period w/ state $(e,k_-)$ is given by:

    \begin{align*}
        V_t(e, k_-)               & = \max_{c,\,k}  u(c) + \beta \sum_{e'} V_{t+1}(e', k) \, P(e,e') \\
        \text{s.t.} \qquad  c + k & = (1 + r_t)k_- + w_t e n                                         \\
        k                         & \geq 0
    \end{align*}

    \begin{itemize}
        \item Denote by $c^*_t(e, k_-)$ and $k^*_t(e, k_-)$ the policy functions that solve the Bellman equation.
        \item Also denote by $D_t(e, K_-)$ the measure of households in state $e$ that own capital in a set $K_-$ at the start of date $t$.
    \end{itemize}
\end{frame}


\begin{frame}{Example: The Krusell-Smith Model}{Household Problem}
    The distribution $D_t$ has law of motion
    \[
        D_{t+1}(e', K) = D_t(e, k^{*-1}_t(e, K)) P(e,e').
    \]
    where $k^{*-1}_t(e, K)$ denotes the inverse of $k^*_t(e, K)$.

    \begin{itemize}
        \item We assume that prior to $t = 0$, the economy is in a steady state.
        \item We suppose that agents start in this stationary distribution at date $0$, so that $D_0 = D_{ss}$.
    \end{itemize}

\end{frame}


\begin{frame}{Example: The Krusell-Smith Model}{Aggregate Capital}
    \begin{itemize}
        \item For any $t$, the policy $k^*_t(e, k_-)$ is a function of the future path
              $\{r_s, w_s\}_{s=t}^{\infty}$.
        \item \alert{Given $D_0 = D_{ss}$, the distribution $D_t(e, K)$ at any $t$ is a function of the entire path $\{r_s, w_s\}_{s=0}^{\infty}$.}
        \item The aggregate household capital holdings are characterized by a capital function $\mathcal{K}_t(\{r_s, w_s\}_{s=0}^{\infty})$.

              \[
                  \mathcal{K}_t\left(\{r_s, w_s\}_{s=0}^{\infty}\right) =
                  \sum_{e} \int_{k_-}
                  k^*_t(e, k_-) D_t(e, dk_-)
              \]
    \end{itemize}

    \begin{alertblock}{Key Insight}
        The ability to reduce interactions between heterogeneous agents to functions such as $\mathcal{K}_t$,
        which map aggregate sequences into aggregate sequences, is key to the sequence-space Jacobian method.
    \end{alertblock}
\end{frame}


\begin{frame}{Example: The Krusell-Smith Model}{Production}
    \begin{itemize}
        \item Firm adopts a Cobb-Douglas production function: \[Y_t = Z_t K_t^{\alpha} N_t^{1-\alpha}\]
        \item The firm’s first-order conditions:
              \begin{align*}
                  r_t & = \alpha Z_t K_{t-1}^{\alpha-1} N_t^{1-\alpha} -\delta \\
                  w_t & = (1 - \alpha) Z_t K_{t-1}^{\alpha} N_t^{-\alpha}
              \end{align*}

    \end{itemize}

    \begin{alertblock}{Key Insight}
        The paths of prices $\{r_t, w_t\}_{t=0}^{\infty}$  are now functions of the exogenous paths $\{Z_t,N_t\}_{t=0}^{\infty}$ and the endogenous path $\{K_t\}_{t=0}^{\infty}$.
    \end{alertblock}
\end{frame}



\begin{frame}{Example: The Krusell-Smith Model}{Capital Market-Clearing Condition}
    Combining together, we can express the capital market-clearing condition at each point in time as a function $H_t$:
    \[H_t(\bm{K},\bm{Z})\equiv\mathcal{K}_t\left(\left\{ \underbrace{\alpha Z_s\left(\frac{K_{s-1}}{\sum\pi(e)en}\right)^{\alpha-1}-\delta}_{r_s}, \underbrace{(1-\alpha)Z_s\left(\frac{K_{s-1}}{\sum\pi(e)en}\right)^\alpha}_{w_s} \right\}_{s=0}^{\infty}\right)-K_t=0\]
    where $\bm{K}$ is the vector of capital holdings $\{K_t\}_{t=0}^{\infty}$ and $\bm{Z}$ is the vector of exogenous shocks $\{Z_t\}_{t=0}^{\infty}$.

    \begin{block}{Equilibrium Capital Path}
        Given initial capital $K_{-1}$ and the exogenous path for productivity $\bm{Z}$ , we have equilibrium path of capital $\{K_t\}_{t=0}^{\infty}$ by \alert{solving the nonlinear equation $H_t(\bm{K},\bm{Z})=0$}.
    \end{block}

\end{frame}


\begin{frame}{Example: The Krusell-Smith Model}{IRF}
    By  \emph{implicit function theorem}, given an exo. shock $\mathrm{d} \bm{Z}= (\mathrm{d} Z_0,\mathrm{d} Z_1...)'$ at $t=0$, the linear impulse response of capital to this shock is given by:
    \[
        d\bm{K} = - H_K^{-1} H_Z \, d\bm{Z}
    \]

    where $H_K$ and $H_Z$ denote the Jacobians of $H$ with respect to $\bm{K}$ and $\bm{Z}$, evaluated at the steady state.
    \begin{itemize}
        \item Given $\mathrm{d}\bm{K}$, the impulse responses of other variables—for example, $\{Y_t, r_t, w_t\}_{t=0}^{\infty}$—follow immediately.

        \item In practice, $H_t(\bm{K},\bm{Z})=0$ is solved up to a sufficiently large horizon $T$ such that $\bm{K}$ and $\bm{Z}$ have approximately returned to their steady states by time $T$.
    \end{itemize}
\end{frame}

\begin{frame}{Example: The Krusell-Smith Model}{Constructing the Jacobian}
    \begin{itemize}
        \item  We use the \emph{chain rule} to relate the Jacobians $H_K$ and $H_Z$ to the derivatives of the $\mathcal{K}$.
        \item For example, differentiating $H_t(\bm{K},\bm{Z})=0$ with respect to $K_s$
        \item     The $(t, s)$ entry of $H_K$ is:
              \[
                  [H_K]_{t,s} = \frac{\partial \mathcal{K}_t}{\partial r_{s+1}} \frac{\partial r_{s+1}}{\partial K_s}
                  + \frac{\partial \mathcal{K}_t}{\partial w_{s+1}} \frac{\partial w_{s+1}}{\partial K_s} - \bm{1}_{\{s = t\}}
              \]

        \item A similar expression applies to $H_Z$.
        \item In addition, the derivatives such as $\frac{\partial r_{s+1}}{\partial K_s}$ at $(K^{ss}, Z^{ss})$ can all be computed analytically.
        \item For example:
              \[
                  \frac{\partial r_{s+1}}{\partial K_s} = \alpha(\alpha - 1) Z^{ss} (K^{ss})^{\alpha-2} (N^{ss})^{1-\alpha}
              \]

    \end{itemize}
\end{frame}



\begin{frame}{Example: The Krusell-Smith Model}{Constructing the Jacobian}
    Recall that:
    \[
        d\bm{K} = - H_K^{-1} H_Z \, d\bm{Z},
    \]
    To get $H_K^{-1} H_Z $, the only thing left is to solve the Jacobians of the $\mathcal{K}$ function with
    respect to its two inputs $r$ and $w$.


    \[
        [H_K]_{t,s} =\alert{ \frac{\partial \mathcal{K}_t}{\partial r_{s+1}} }\frac{\partial r_{s+1}}{\partial K_s}
        + \alert{ \frac{\partial \mathcal{K}_t}{\partial w_{s+1}} } \frac{\partial w_{s+1}}{\partial K_s} - \bm{1}_{\{s = t\}}
    \]

    Let \[\mathcal{J}_{t,s}^{\mathcal{K},r}=\frac{\partial\mathcal{K}_t}{\partial r_s},\quad\mathcal{J}_{t,s}^{\mathcal{K},w}=\frac{\partial\mathcal{K}_t}{\partial w_s}\] be the Jacobians up to some truncation horizon $T$.



\end{frame}


\begin{frame}{Example: The Krusell-Smith Model}{Constructing the Jacobian}
    \begin{alertblock}{Sufficient statistics}
        The Jacobians $\mathcal{J}^{\mathcal{K},r}$ and $\mathcal{J}^{\mathcal{K},w}$ are sufficient statistics for the household in sequence space equilibrium:
        \begin{itemize}
            \item Shocks propagate through the dynamics of the household capital distribution, but\emph{ we do not need those details if we have the Jacobians.}
            \item Column $s$ of $\mathcal{J}^{\mathcal{K},r}$ is \alert{the impulse response of capital to a news shock $dr_s$}; the Jacobian gives these for all $s=0,\ldots,T-1$.
            \item These are rich objects, but \emph{there is a fast new algorithm to compute them.}
        \end{itemize}
    \end{alertblock}
\end{frame}


\begin{frame}{Fake News Algorithm}{General Model Representation}

    We represent a heterogeneous-agent problem as a mapping from a time path of
    aggregate inputs $\{\bm{X}_t\}$ to a time path of aggregate outputs $\{\bm{Y}_t\}$.

    \vspace{0.5em}

    Assume:
    \begin{itemize}
        \item $\bm{X}_t \in \mathbb{R}^{n_x}$: aggregate inputs,
        \item $\bm{Y}_t \in \mathbb{R}^{n_y}$: aggregate outputs,
        \item the distribution is discretized on $n_g$ grid points,
        \item $\bm{D}_t \in \mathbb{R}^{n_g}$: cross-sectional distribution of agents at time $t$.
    \end{itemize}

    \vspace{0.5em}

    Aggregate outputs are averages of individual outcomes against the distribution:
    \[
        \bm{Y}_t = \bm{y}_t' \bm{D}_t,
    \]
    where $\bm{y}_t$ is an $n_g \times n_y$ matrix of individual outcomes
    (for example, consumption, savings, labor supply, or other individual-level objects).
\end{frame}


\begin{frame}{Fake News Algorithm}{Three Core Equations}

    We assume there exist three functions $v$, $\Lambda$, and $y$ such that,
    for a given initial distribution $\bm{D}_0$, the aggregate sequence solves
    \[
        \bm{v}_t = v(\bm{v}_{t+1}, \bm{X}_t),
    \]
    \[
        \bm{D}_{t+1} = \Lambda(\bm{v}_{t+1}, \bm{X}_t)\,\bm{D}_t,
    \]
    \[
        \bm{Y}_t = y(\bm{v}_{t+1}, \bm{X}_t)' \bm{D}_t.
    \]

    \vspace{0.7em}

    Interpretation:
    \begin{itemize}
        \item $\bm{v}_t$: backward-looking object, typically the value function;
        \item $\Lambda(\bm{v}_{t+1}, \bm{X}_t)$: transition matrix governing the law of motion of the distribution;
        \item $y(\bm{v}_{t+1}, \bm{X}_t)$: individual outcomes used to aggregate into $\bm{Y}_t$.
    \end{itemize}
\end{frame}

% ------------------------------------------------
\begin{frame}{Fake News Algorithm}{Economic Meaning of the System}
    \small
    \begin{block}{Step 1: Backward}
        Given the future object $\bm{v}_{t+1}$ and current aggregates $\bm{X}_t$,
        solve for the current object $\bm{v}_t$.
    \end{block}

    \begin{block}{Step 2: Forward}
        Use $\Lambda(\bm{v}_{t+1}, \bm{X}_t)$ to update the cross-sectional distribution:
        \[
            \bm{D}_{t+1} = \Lambda(\bm{v}_{t+1}, \bm{X}_t)\bm{D}_t.
        \]
    \end{block}

    \begin{block}{Step 3: Aggregate}
        Aggregate individual outcomes across agents:
        \[
            \bm{Y}_t = y(\bm{v}_{t+1}, \bm{X}_t)' \bm{D}_t.
        \]
    \end{block}


\end{frame}

% ------------------------------------------------
\begin{frame}{Fake News Algorithm}{Steady State and Transition Paths}
    \small
    For a constant aggregate input $\bm{X}_{ss}$, the steady state is a fixed point
    \[
        (\bm{Y}_{ss}, \bm{v}_{ss}, \bm{D}_{ss})
    \]
    of the system.

    \vspace{0.5em}

    For convenience, define
    \[
        \Lambda_{ss} \equiv \Lambda(\bm{v}_{ss}, \bm{X}_{ss}),
        \qquad
        y_{ss} \equiv y(\bm{v}_{ss}, \bm{X}_{ss}).
    \]

    \vspace{0.5em}

    We consider transition paths of length $T$ that end at the steady state:
    \[
        \bm{X}_{T-1} = \bm{X}_{ss},
        \qquad
        \bm{v}_T = \bm{v}_{ss}.
    \]

    The initial distribution $\bm{D}_0$ is given and often set equal to $\bm{D}_{ss}$.
    \vspace{0.5em}
    Hence, this framework is designed to study \alert{transitory shocks around a steady state}.
\end{frame}
